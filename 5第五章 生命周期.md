# 📱 iOS 生命周期


```swift
class AppDelegate: UIResponder, UIApplicationDelegate {

    // 程序启动入口 只在应用程序启动时执行一次
    func application(_ application: UIApplication, didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?) -> Bool {
        print("应用程序启动")
        return true
    }

    // MARK: UISceneSession Lifecycle
    // 创建新场景会话会调用
    func application(_ application: UIApplication, configurationForConnecting connectingSceneSession: UISceneSession, options: UIScene.ConnectionOptions) -> UISceneConfiguration {
        print("创建新场景会话")
        return UISceneConfiguration(name: "Default Configuration", sessionRole: connectingSceneSession.role)
    }
    
    // 用户放弃场景会话会调用，释放被丢弃的场景资源
    func application(_ application: UIApplication, didDiscardSceneSessions sceneSessions: Set<UISceneSession>) {
        print("放弃场景会话")
    }
}

class SceneDelegate: UIResponder, UIWindowSceneDelegate {

    var window: UIWindow?

    // 场景代理文件用来管理应用程序的多个场景，以及管理用户界面的生命周期
    func scene(_ scene: UIScene, willConnectTo session: UISceneSession, options connectionOptions: UIScene.ConnectionOptions) {
        guard let _ = (scene as? UIWindowScene) else { return }
        print("场景载入后需要执行的代码")
    }
    
    // 场景被系统释放时调用，在场景进入背景后不久或会员被丢弃时发生
    func sceneDidDisconnect(_ scene: UIScene) {
        print("场景被系统释放时调用")
    }
    
    // 场景从非活动状态转换为活动状态时调用，可在这个方法中重启场景处于非活动状态时已暂停任务
    func sceneDidBecomeActive(_ scene: UIScene) {
        print("场景从非活动状态转换为活动状态")
    }
    
    // 场景从活动状态转换为非活动状态时调用，可能是临时中断而发生，例如打电话
    func sceneWillResignActive(_ scene: UIScene) {
        print("场景从活动状态转换为非活动状态")
    }
    
    // 场景从背景过度到前景时调用，可撤销在进入背景时所做的改变
    func sceneWillEnterForeground(_ scene: UIScene) {
        print("场景从背景过度到前景")
    }
    
    // 场景从前景过度到背景时调用，可使用此方法保存数据，释放共享资源或存储场景的状态信息
    func sceneDidEnterBackground(_ scene: UIScene) {
        (UIApplication.shared.delegate as? AppDelegate)?.saveContext()
        print("场景从前景过度到背景")
    }
}
```

1. 首次启动
- 应用程序启动
- 场景载入后需要执行的代码
- 场景从背景过度到前景
- 场景从非活动状态转换为活动状态

2. 回主界面
- 场景从活动状态转换为非活动状态
- 场景从前景过度到背景

3. 点击app
- 场景从背景过度到前景
- 场景从非活动状态转换为活动状态

4. 双击home 打开运行程序
- 场景从活动状态转换为非活动状态

5. 清理运行程序
- 场景从前景过度到背景
- 场景被系统释放时调用


### AppDelegate
应用代理文件，是系统运行本应用的委托，定义程序的进入与退出、设备方向的旋转等众多全局方法

### SceneDelegate
场景代理文件，管理应用的场景及用户界面的生命周期

### ViewController
视图控制器，可创建和管理视图，也可检测设备方向变化，并调整视图大小以适应屏幕

### Main
故事板，可使所有视图控制器以及他们之间的关系一目了然

### Assets
集中管理项目图片、颜色、符号、数据

### LaunchScreen
启动场景故事板，可帮助设计和适配程序启动页

### Info
每个程序都会使用信息属性列表文件，存储项目配置信息
